5.物流单导入

导入（注意核对条数！！！）

按照省，市，首重重量，首重金额，续重重量，续重金额导入LOGISTIC_IMPORT_TABLE，注意！！！删除旧数据

填写省（注意省代码导入后应该不为空！！！！）

merge into LOGISTIC_IMPORT_TABLE a

using (
        SELECT
            PROVINCE_ID,
            LOGISTIC_IMPORT_TABLE.o_province_name
        FROM
            PROVINCE,
            LOGISTIC_IMPORT_TABLE
        WHERE
            PROVINCE_NAME LIKE CONCAT (
                LOGISTIC_IMPORT_TABLE.o_province_name,
                '%'
            ) 
    ) b

on(a.o_province_name=b.o_province_name)  

when matched then update set a.sys_province_id=b.PROVINCE_ID;

填写市（注意市代码导入后应该不为空！！！！）

merge into LOGISTIC_IMPORT_TABLE a

using (
        SELECT
            CITY_ID,
            LOGISTIC_IMPORT_TABLE.o_city_name
        FROM
            CITY,
            LOGISTIC_IMPORT_TABLE
        WHERE
            LOGISTIC_IMPORT_TABLE.o_city_name is not null
            and CITY_NAME LIKE CONCAT (
                LOGISTIC_IMPORT_TABLE.o_city_name,
                '%'
            ) 
    ) b

on(a.o_city_name=b.o_city_name)  

when matched then update set a.sys_city_id=b.CITY_ID;
